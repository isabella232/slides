PubNub Data Channels

// Listen for events
PUBNUB.subscribe({
  channel: "my_channel",
  message: function (message) {
    console.log(message);
  }
});

// Send Events
PUBNUB.publish({
  channel: "my_channel",
  message: "Hello World!"
});
